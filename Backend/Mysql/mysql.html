# node 中使用 mysql

# 连接数据库

// ./mysql文件
const mysql = require("mysql");
let Mysql = (sql) => {
  return new Promise(function(resolve, reject) {
    // 连接数据库
    const connection = mysql.createConnection({
      host: "8.131.83.251", //地址
      user: "root", //用户名
      password: "123abc", // 数据库密码
      port: "3306", // 端口号
      database: "drugstore", //数据库名
    });
    // 连接数据库，判断是否连接成功
    connection.connect(function(err) {
      if (err) reject([]);
    });
    connection.query(sql, function(err, result) {
      if (err) reject([]);
      resolve(result);
    });
    connection.end();
  });
};
module.exports = Mysql;
