# 生成 git 密钥 sshkey

ssh-keygen -t rsa -C "自己邮箱"
输入完成按三下回车
1
2

从.ssh\id_rsa.pub 文件里复制公钥
经过上面两步，我们打开 C:\Users\xxx\.ssh 就看到了三个文件‘id_rsa.pub’、‘id_rsa’‘known_hosts’，分别是公钥，私钥，记录的操作仓库地址； 这其中私钥是我们本机的【千万不能给公开】，公钥是给我们用来和码云链接用的，known_hosts 没啥用，留着别删。 id_rsa.pub 右键记事本打开，ctrl+A 全选后立刻关闭该文件，切记，千万不可以修改这里的内容

# 托管项目

在gitee或者github上面创建一个空白项目使用git clone "克隆地址" 在本地创建一个项目把该项目中的git文件用克隆下来的git文件替换掉

# git 常用命令

# 1. 添加当前目录的所有文件到暂存区

git add .
1

# 2. 提交暂存区到仓库区

git commit -m "提交说明"  //（""内空白会失败）
1

# 3. 查看仓库当前的状态

git status
1

# 4. 提交代码到远程仓库（主分支）最终的提交

git push origin master 或者直接git push
1

# 分支命令

# 1. 创建分支

git branch name
1

# 2. 切换分支

git checkout name
1

# 3. 查看分支

git branch
git branch -a // 查看所有分支
1
2

# 4. 查看远程仓库地址

git branch -v
1

# 5. 删除本地分支

git branch -d devName
1

# 6. 删除连接的远程本地分支

git branch -r -d origin/branch-name
1

# 7. 删除远程分支

git push origin --delete [分支的名字]
1

# 8. 同时删除本地和远程分支

git push -d origin [分支的名字]
1

# 9. 强制推送本地分

git push --set-upstream origin [分支的名字]
1
